# Python code to calculate the first 10,000 terms of the series and multiply by 4

def calculate_series(terms):
    total = 0.0
    for n in range(terms):
        term = (-1) ** n / (2 * n + 1)  # Alternating series term
        total += term
    result = total * 4  # Multiply by 4
    return result

# Calculate the first 10,000 terms
result = calculate_series(10000)
print(result)

Output:
